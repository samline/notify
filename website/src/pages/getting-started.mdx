# Getting Started

Samline Sonner is a shared toast runtime for React, Vue, Svelte, vanilla JS and browser/CDN usage.

<Steps>
### Install

<Tabs items={['pnpm', 'npm', 'yarn', 'bun']}>


  </Tab>

### Choose a runtime

<Tabs items={['Vanilla', 'React', 'Vue', 'Svelte', 'Browser']}>
  <Tab>
    ```ts
    import { createToaster, toast } from '@samline/sonner';

    createToaster({ position: 'bottom-right' });

    document.querySelector('button')?.addEventListener('click', () => {
      toast('Hello from vanilla');
    });
    ```
  </Tab>
  <Tab>
    ```tsx
    import { Toaster, toast } from '@samline/sonner/react';

    export default function App() {
      return (
        <>
          <Toaster />
          <button onClick={() => toast('Hello from React')}>Toast</button>
        </>
      );
    }
    ```
  </Tab>
  <Tab>
    ```ts
    import { createApp } from 'vue';
    import { SonnerPlugin, Toaster, toast } from '@samline/sonner/vue';

    const app = createApp({
      components: { Toaster },
      template: `<><Toaster /><button @click="notify">Toast</button></>`,
      methods: {
        notify() {
          toast('Hello from Vue');
        },
      },
    });

    app.use(SonnerPlugin);
    app.mount('#app');
    ```
  </Tab>
  <Tab>
    ```svelte
    <script lang="ts">
      import { toaster, toast } from '@samline/sonner/svelte';

      const options = { position: 'bottom-right' };
    </script>

    <span use:toaster={options} hidden aria-hidden="true"></span>
    <button on:click={() => toast('Hello from Svelte')}>Toast</button>
    ```
  </Tab>
  <Tab>
    ```html
    <script type="module">
      import '@samline/sonner/browser';

      window.Sonner.toast('Hello from browser');
    </script>
    ```
  </Tab>
</Tabs>

### React Toaster

It can be placed anywhere, even in server components such as `layout.tsx`.

```tsx
import { Toaster } from '@samline/sonner/react';

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <body>
        {children}
        <Toaster />
      </body>
    </html>
  );
}
```

### Render a toast

```tsx
import { toast } from '@samline/sonner/react';

function MyToast() {
  return <button onClick={() => toast('This is a sonner toast')}>Render my toast</button>;
}
```
